SELECT order_id, product_id, add_to_cart_order, reordered

    o.order_id,
    o.user_id,
    o.order_number,
    o.order_dow,
    o.order_hour_of_day,
    o.days_since_prior_order,

  ON op.product_id = p.product_id

  ON p.aisle_id = a.aisle_id
INNER JOIN departments d
  ON p.department_id = d.department_id;
""")

orders = pd.read_sql_query("SELECT * FROM instacart_wide;", con)

Field Dictionary¶

Field Type Description
order_id int Unique identifier for an order placed by a user.
user_id int Unique identifier for the customer who placed the order.
order_number int Sequence number of the order for a given user (1 = first order, 2 = second order, etc.).
order_dow int (0–6) Day of the week the order was placed (0 = Sunday, …, 6 = Saturday).
order_hour_of_day int (0–23) Hour of the day the order was placed (24-hour format).
days_since_prior_order float Number of days since the user’s previous order (NaN for the user’s first order).
product_id int Unique identifier for a product in the catalog.
add_to_cart_order int Sequence in which the product was added to the cart within the order.
reordered int (0/1) Flag indicating if the product was ordered by the user before (1 = repeat, 0 = first-time).
aisle_id int Numeric identifier for the aisle the product belongs to.
department_id int Numeric identifier for the department the product belongs to.
product_name object Human-readable product name (e.g., Organic Bananas).
aisle object Human-readable aisle name (e.g., fresh fruits, yogurt).
department object Human-readable department name (e.g., produce, dairy eggs).

First, let's separate the last order of each user from the rest (the history). This will be used as our target.¶

In [21]:
def split_history_target(df: pd.DataFrame):
    last_orders = df.groupby("user_id")["order_number"].max().rename("target_order_number").reset_index()
    mark = df.merge(last_orders, on="user_id", how="left")

    history = mark[mark["order_number"] < mark["target_order_number"]]
    target_orders = mark[mark["order_number"] == mark["target_order_number"]]

    return history, target_orders

Now, let's make our history orders our X, and make all products seen in history for each user that are in the last order as our Y.¶

In [22]:
def build_X_y(df: pd.DataFrame):
    history, target_orders = split_history_target(df)

    X = history.copy()

    next_purchased_products = (target_orders[["user_id", "product_id"]].drop_duplicates())
    next_purchased_products["in_next_order"] = 1

    y = (
        X[["user_id", "product_id"]]
        .merge(next_purchased_products, on=["user_id", "product_id"], how="left")
        ["in_next_order"]
        .fillna(0)
        .astype("int8")
        .to_frame(name="in_next_order")   
    )

    return X, y

Pipeline¶

The fields order_dow (day of week) and order_hour_of_day (hour of day) are cyclical in nature. To capture this, I applied cyclical encoding using sine and cosine transformations. This preserves the natural ordering and proximity of values (e.g., hour 23 is close to hour 0) by mapping them onto a circle.

In [24]:
class DropColumns(BaseEstimator, TransformerMixin):
    def __init__(self, cols):
        self.cols = cols
    def fit(self, X, y=None):
        return self
    def transform(self, X):
        return X.drop(columns=[c for c in self.cols if c in X.columns], errors="ignore")

class CyclicalTimeEncoder(BaseEstimator, TransformerMixin):
    def fit(self, X, y=None): 
        return self
    def transform(self, X):
        X = X.copy()
        X["order_dow_sin"]  = np.sin(2 * np.pi * X["order_dow"] / 7)
        X["order_dow_cos"]  = np.cos(2 * np.pi * X["order_dow"] / 7)
        X["order_hour_sin"] = np.sin(2 * np.pi * X["order_hour_of_day"] / 24)
        X["order_hour_cos"] = np.cos(2 * np.pi * X["order_hour_of_day"] / 24)
        return X

class CartRelativePosition(BaseEstimator, TransformerMixin):
    def fit(self, X, y=None):
        return self
    def transform(self, X):
        X = X.copy()
        cart_size = X.groupby("order_id")["add_to_cart_order"].transform("max")
        X["add_to_cart_rel"] = X["add_to_cart_order"] / cart_size
        X = X.drop(columns=["add_to_cart_order"], errors="ignore")
        return X
    
class ProductIDCount(BaseEstimator, TransformerMixin):
    def fit(self, X, y=None):
        self.counts_ = X["product_id"].value_counts(dropna=False).to_dict()
        return self
    def transform(self, X):
        X = X.copy()
        cnt = X["product_id"].map(self.counts_).fillna(0).astype(np.float32)
        X["product_id_count"] = np.log1p(cnt) 
        X = X.drop(columns=["product_id"], errors="ignore")
        return X
    
class FillNa(BaseEstimator, TransformerMixin):
    def __init__(self, cols, value=0.0):
        self.cols = cols
        self.value = value
    def fit(self, X, y=None):
        return self
    def transform(self, X):
        X = X.copy()
        for c in self.cols:
            if c in X.columns:
                X[c] = X[c].fillna(self.value)
        return X

def build_mlp(input_dim, hidden_units=(128,64), dropout=0.2, lr=5e-4):
    inputs = keras.Input(shape=(input_dim,))
    x = inputs
    for h in hidden_units:
        x = keras.layers.Dense(h, kernel_initializer="he_normal",
                               kernel_regularizer=keras.regularizers.l2(1e-6))(x)
        x = keras.layers.BatchNormalization()(x)
        x = keras.layers.Activation("relu")(x)
        x = keras.layers.Dropout(dropout)(x)
    outputs = keras.layers.Dense(1, activation="sigmoid")(x)
    model = keras.Model(inputs, outputs)
    model.compile(
        optimizer=keras.optimizers.Adam(learning_rate=lr),
        loss="binary_crossentropy",
        metrics=[
            keras.metrics.BinaryAccuracy(name="accuracy"),
            keras.metrics.AUC(name="auc"),
            keras.metrics.AUC(curve="PR", name="pr_auc"),
        ],
    )
    return model
